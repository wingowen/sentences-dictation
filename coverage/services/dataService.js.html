/**
 * 获取树节点的扁平列表（用于查找）
 * @param {Array} tree - 树形配置
 * @returns {Array} 扁平节点列表
 */
export const flattenTree = (tree) => {
  const result = [];
  const traverse = (nodes) => {
    for (const node of nodes) {
      result.push(node);
      if (node.children) {
        traverse(node.children);
      }
    }
  };
  traverse(tree);
  return result;
};
